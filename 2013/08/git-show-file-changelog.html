Date:   Sun Jun 23 15:36:58 2013 +0800

    commit title

diff --git a/fabfile.py b/fabfile.py
index 4ed36fa..1408118 100644
--- a/fabfile.py
+++ b/fabfile.py
@@ -9,28 +9,29 @@ from fabric.api import settings

 @task
 def make_html():
-    """ 生成  html  文件 """
-    local('pelican content -o output -s pelicanconf.py')
